Users Route

/users

//Get user profile
GET{
"sessionToken": "String",
}

//Update a user account
PUT{
"sessionToken": "String",
"name": "String", (optional)
"email": "String", (optional)
"password": "String", (optional)
}

//Delete a user account - FUTURE
DELETE{
"sessionToken": "String",
}
