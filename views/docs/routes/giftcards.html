Giftcards Route

/giftcards

//Create a giftcard
POST{
"sessionToken": "String",
"toName": "String", (recipient name)
"fromName": "String", (recipient name)
"phone": "String", (recipient phone, plain integer string)
"email": "String", (sender email)
"amount": "String", (0-50000 in cents)
"iconId": "String",
"locationId": "String",
"message": "String",
"stripeCardToken": "String",
"sendDate": "String" (optional)
}

//Get all giftcards
GET{
"sessionToken": "String",
}

//Individual giftcard operations

/giftcards/:id

//Get a giftcard
GET{
"sessionToken": "String",
}

//Update a giftcard - FUTURE
PUT{
"sessionToken": "String",
"iconId": "String",
"message": "String",
}

/giftcards/later

//Save giftcard for later and send email link
POST{
"sessionToken": "String",
"giftcardId": "String" (user must already have an email associated with acct)
}
